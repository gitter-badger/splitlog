45      public void testGetMessages() {
46          final String message1 = "test";
47          final String message2part1 = "test1";
48          final String message2part2 = "test2";
49          final String message3part1 = "test3";
50          final String message3part2 = "test4";
51          final LogTailer tailer = this.getLogWatch().startTailing();
52          // test simple messages
53          String result = this.getWriter().write(message1, tailer);
54          Assert.assertEquals(message1, result);
55          result = this.getWriter().write(message1, tailer);
56          Assert.assertEquals(message1, result);
57          result = this.getWriter().write(message2part1 + "\n" + message2part2, tailer);
58          Assert.assertEquals(message2part2, result);
59          result = this.getWriter().write(message3part1 + "\r\n" + message3part2, tailer);
60          Assert.assertEquals(message3part2, result);
61          // now validate the results
62          final List<Message> messages = tailer.getMessages();
63          Assert.assertEquals(5, messages.size());
64          Assert.assertEquals(message1, messages.get(1).getLines().get(0));
65          Assert.assertEquals(message2part1, messages.get(2).getLines().get(0));
66          Assert.assertEquals(message2part2, messages.get(3).getLines().get(0));
67          Assert.assertEquals(message3part1, messages.get(4).getLines().get(0));
68          // final part of the message, message3part2, will remain unflushed
69          this.getLogWatch().terminateTailing(tailer);
70          Assert.assertTrue(tailer.isTerminated());

130     @Test
131     public void testNesting() {
132         final String message1 = "test1";
133         final String message2 = "test2";
134         final String message3 = "test3";
135         final String message4 = "test4";
136         final String message5 = "test5";
137         final LogTailer tailer = this.getLogWatch().startTailing();
138         // make sure the messages are received by the first tailer
139         this.getWriter().write(message1, tailer);
140         String result = this.getWriter().write(message2, tailer);
141         Assert.assertEquals(message2, result);
142         Assert.assertEquals(1, tailer.getMessages().size());
143         // start a second tailer, send some messages
144         final LogTailer nestedTailer = this.getLogWatch().startTailing();
145         result = this.getWriter().write(message3, tailer);
146         result = this.getWriter().write(message4, tailer);
147         Assert.assertEquals(message4, result);
148         this.getLogWatch().terminateTailing(nestedTailer);
149         // send another message, so the original tailer has something extra
150         Assert.assertEquals(true, nestedTailer.isTerminated());
151         result = this.getWriter().write(message5, tailer);
152         // and make sure that the original tailer has all messages
153         Assert.assertEquals(4, tailer.getMessages().size());
154         Assert.assertEquals(message1, tailer.getMessages().get(0).getLines().get(0));
155         Assert.assertEquals(message2, tailer.getMessages().get(1).getLines().get(0));
156         Assert.assertEquals(message3, tailer.getMessages().get(2).getLines().get(0));
157         Assert.assertEquals(message4, tailer.getMessages().get(3).getLines().get(0));
158         // and the nested tailer has only the two while it was running
159         Assert.assertEquals(2, nestedTailer.getMessages().size());
160         Assert.assertEquals(message2, nestedTailer.getMessages().get(0).getLines().get(0));
161         Assert.assertEquals(message3, nestedTailer.getMessages().get(1).getLines().get(0));
162     }

172         Assert.assertFalse("Tailer terminated without termination.", this.getLogWatch().isTerminated(tailer2));
173         Assert.assertTrue("Tailer not terminated after termination.", this.getLogWatch().isTerminated(tailer1));
174         Assert.assertTrue("Wrong termination result.", this.getLogWatch().terminateTailing());
175         Assert.assertFalse("Wrong termination result.", this.getLogWatch().terminateTailing());
176         Assert.assertTrue("Tailer not terminated after termination.", this.getLogWatch().isTerminated(tailer2));
177         Assert.assertTrue("Log watch not terminated after termination.", this.getLogWatch().isTerminated());
