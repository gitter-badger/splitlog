78      @Test
79      public void testNesting() {
80          final String message1 = "test1";
81          final String message2 = "test2";
82          final String message3 = "test3";
83          final String message4 = "test4";
84          final String message5 = "test5";
85          final Follower follower = this.getLogWatch().follow();
86          // make sure the messages are received by the first follower
87          this.getWriter().write(message1, follower);
88          String result = this.getWriter().write(message2, follower);
89          Assert.assertEquals(message2, result);
90          Assert.assertEquals(1, follower.getMessages().size());
91          // start a second follower, send some messages
92          final Follower nestedTailer = this.getLogWatch().follow();
93          result = this.getWriter().write(message3, follower);
94          result = this.getWriter().write(message4, follower);
95          Assert.assertEquals(message4, result);
96          this.getLogWatch().unfollow(nestedTailer);
97          // send another message, so the original follower has something extra
98          Assert.assertFalse(nestedTailer.isFollowing());
99          result = this.getWriter().write(message5, follower);
100         // and make sure that the original follower has all messages
101         Assert.assertEquals(4, follower.getMessages().size());
102         Assert.assertEquals(message1, follower.getMessages().get(0).getLines().get(0));
103         Assert.assertEquals(message2, follower.getMessages().get(1).getLines().get(0));
104         Assert.assertEquals(message3, follower.getMessages().get(2).getLines().get(0));
105         Assert.assertEquals(message4, follower.getMessages().get(3).getLines().get(0));
106         // and the nested follower has only the two while it was running
107         Assert.assertEquals(2, nestedTailer.getMessages().size());
108         Assert.assertEquals(message2, nestedTailer.getMessages().get(0).getLines().get(0));
109         Assert.assertEquals(message3, nestedTailer.getMessages().get(1).getLines().get(0));
110     }

115         final Follower follower1 = this.getLogWatch().follow();
116         Assert.assertTrue("Follower terminated immediately after starting.", this.getLogWatch().isFollowedBy(follower1));
117         final Follower follower2 = this.getLogWatch().follow();
118         Assert.assertTrue("Wrong termination result.", this.getLogWatch().unfollow(follower1));
119         Assert.assertFalse("Wrong termination result.", this.getLogWatch().unfollow(follower1));
120         Assert.assertTrue("Follower terminated without termination.", this.getLogWatch().isFollowedBy(follower2));
121         Assert.assertFalse("Follower not terminated after termination.", this.getLogWatch().isFollowedBy(follower1));
122         Assert.assertTrue("Wrong termination result.", this.getLogWatch().terminate());
123         Assert.assertFalse("Wrong termination result.", this.getLogWatch().terminate());
124         Assert.assertFalse("Follower not terminated after termination.", this.getLogWatch().isFollowedBy(follower2));
125         Assert.assertTrue("Log watch not terminated after termination.", this.getLogWatch().isTerminated());
