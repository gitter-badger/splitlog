26      @Test
27      public void testMergeWithDifferentWatches() {
28          // write into first file
29          final LogWatch watch1 = LogWatchBuilder.getDefault().watchedFile(LogWriter.createTempFile())
30                  .build();
31          final Follower follower1 = watch1.startFollowing();
32          LogWriter.write(follower1, MergingTest.MESSAGE_1);
33          // write into second file
34          final LogWatch watch2 = LogWatchBuilder.getDefault().watchedFile(LogWriter.createTempFile())
35                  .build();
36          final Follower follower2 = watch2.startFollowing();
37          LogWriter.write(follower2, MergingTest.MESSAGE_2);
38          // merge both
39          final MergingFollower merge = follower2.mergeWith(follower1);
40          Assertions.assertThat(merge.getMerged()).containsOnly(follower1, follower2);
41          Assertions.assertThat(merge.getMessages()).hasSize(0); // zero ACCEPTED
42          LogWriter.write(follower2, MergingTest.MESSAGE_3);
43          Assertions.assertThat(merge.getMessages()).hasSize(1); // one ACCEPTED
44          LogWriter.write(follower1, MergingTest.MESSAGE_4);
45          // third follows the first file; merge should have the message just once
46          final Follower follower3 = watch1.startFollowing();
47          final MergingFollower merge2 = follower3.mergeWith(merge);
48          Assertions.assertThat(merge.getMerged()).containsOnly(follower1, follower2);
49          Assertions.assertThat(merge2).isNotSameAs(merge);
50          Assertions.assertThat(merge2.getMerged()).containsOnly(follower1, follower2, follower3);
51          LogWriter.write(follower3, MergingTest.MESSAGE_5);
52          Assertions.assertThat(merge.getMessages()).hasSize(3);
53          Assertions.assertThat(merge2.getMessages()).hasSize(3);
54          // test writing
55          final List<String> messages = new LinkedList<String>();
56          messages.add(MergingTest.MESSAGE_1);
57          messages.add(MergingTest.MESSAGE_2);
58          messages.add(MergingTest.MESSAGE_4);
59          try {
60              final File f = LogWriter.createTempFile();
61              merge2.write(new FileOutputStream(f), NoopMessageFormatter.INSTANCE);
62              Assertions.assertThat(f).exists();
63              final List<String> lines = FileUtils.readLines(f, "UTF-8");
64              Assertions.assertThat(lines).isEqualTo(messages);
65          } catch (final Exception e) {
66              Assertions.fail("Couldn't write to file.");
67          }
68          // and now separate the second, making MESSAGE_2 disappear
69          final MergingFollower merge3 = merge2.remove(follower2);
70          Assertions.assertThat(merge2.getMerged()).contains(follower2);
71          Assertions.assertThat(merge3.getMerged()).doesNotContain(follower2);
72          Assertions.assertThat(merge3.getMessages()).hasSize(2);
73          // now only MESSAGE_4 remains ACCEPTED
74          final MergingFollower merge4 = merge3.remove(follower1);
75          Assertions.assertThat(merge2.getMerged()).contains(follower1);
76          Assertions.assertThat(merge4.getMerged()).doesNotContain(follower1);
77          Assertions.assertThat(merge4.getMessages()).hasSize(1);
78          // and now stop the followers to see what happens to the merges
79          follower3.stop();
80          Assertions.assertThat(follower3.isStopped()).isTrue();
81          Assertions.assertThat(merge4.isStopped()).isTrue();
82          Assertions.assertThat(merge3.isStopped()).isFalse();
83          Assertions.assertThat(merge2.isStopped()).isFalse();
84          Assertions.assertThat(merge.isStopped()).isFalse();
85          // first watch is terminated, second one remains alive
86          watch1.stop();
87          Assertions.assertThat(follower1.isStopped()).isTrue();
88          Assertions.assertThat(merge4.isStopped()).isTrue();
89          Assertions.assertThat(merge3.isStopped()).isTrue();
90          Assertions.assertThat(merge2.isStopped()).isFalse();
91          Assertions.assertThat(follower2.isStopped()).isFalse();
92          Assertions.assertThat(merge.isStopped()).isFalse();
93          merge.stop();
94          Assertions.assertThat(merge.isStopped()).isTrue();
95          Assertions.assertThat(merge2.isStopped()).isTrue();
96          Assertions.assertThat(follower2.isStopped()).isTrue();
97      }

99      @Test
100     public void testMergeWithSameLogWatch() {
101         final Follower f = this.getLogWatch().startFollowing();
102         final Follower f2 = this.getLogWatch().startFollowing();
103         final MergingFollower mf = f.mergeWith(f2);
104         // send the message; the merged should still contain only the one
105         LogWriter.write(f, "test");
106         LogWriter.write(f2, "test2");
107         Assertions.assertThat(mf.getMessages()).hasSize(1);
108         // add third follower, will only receive the second message
109         final Follower f3 = this.getLogWatch().startFollowing();
110         LogWriter.write(f3, "test3");
111         final MergingFollower mf2 = mf.mergeWith(f3);
112         Assertions.assertThat(mf2).isNotEqualTo(mf);
113         Assertions.assertThat(mf.getMessages()).hasSize(2);
114         Assertions.assertThat(mf2.getMessages()).hasSize(2);
115         // remove both followers from first merge, verify results
116         final MergingFollower mf3 = mf.remove(f);
117         Assertions.assertThat(mf3.getMessages()).hasSize(2);
118         Assertions.assertThat(mf3.isStopped()).isFalse();
119         Assertions.assertThat(this.getLogWatch().stopFollowing(f2)).isTrue();
120         Assertions.assertThat(mf3.isStopped()).isTrue(); // no followers are
121         // following
122         Assertions.assertThat(mf.getMessages()).hasSize(2);
123         Assertions.assertThat(mf3.remove(f2)).isEqualTo(null);
124         // none of these changes should have affected the second merge
125         Assertions.assertThat(mf2.getMessages()).hasSize(2);
126         final MergingFollower mf5 = mf2.remove(f2);
127         this.getLogWatch().stopFollowing(f);
128         Assertions.assertThat(mf5.getMessages()).hasSize(2);
129         final MergingFollower mf6 = mf5.remove(f);
130         Assertions.assertThat(mf6.getMessages()).hasSize(1);
131         this.getLogWatch().stop();
132         Assertions.assertThat(mf.isStopped()).isTrue();
133     }
