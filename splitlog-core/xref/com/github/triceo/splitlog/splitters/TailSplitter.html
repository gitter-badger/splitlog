11  /**
12   * The purpose of classes implementing this interface is to interpret the log
13   * files. Each implementation should be familiar with one type of log file (i.e.
14   * {@link JBossServerLogTailSplitter} for JBoss server logs) and be able to
15   * interpret the messages and their lines.
16   * 
17   */
18  public interface TailSplitter {
19  
20      /**
21       * Whether or not this particular line from the log starts a new log
22       * message. If so, it will be used to trigger a {@link Message} being
23       * produced and possibly stored.
24       * 
25       * @param line
26       *            Line from the log.
27       * @return True if this is the first line of a new log message.
28       */
29      boolean isStartingLine(final String line);
30  
31      /**
32       * Read the message and find the date when the message was submitted.
33       * 
34       * @param message
35       *            Message in question.
36       * @return Date from the message, or the current date if not found.
37       */
38      Date determineDate(final MessageBuilder message);
39  
40      /**
41       * Read the message and try to identify an exception stack trace within.
42       * 
43       * @param message
44       *            Message in question.
45       * @return Exception data if found, null otherwise.
46       */
47      ExceptionDescriptor determineException(final MessageBuilder message);
48  
49      /**
50       * Read the message and try to find its severity.
51       * 
52       * @param message
53       *            Message in question.
54       * @return Severity included in the message, {@link MessageSeverity#UNKNOWN}
55       *         otherwise.
56       */
57      MessageSeverity determineSeverity(final MessageBuilder message);
58  
59      /**
60       * Read the message and try to find its type.
61       * 
62       * @param message
63       *            Message in question.
64       * @return Type guessed from the message, {@link MessageType#LOG} if
65       *         undetermined.
66       */
67      MessageType determineType(final MessageBuilder message);
68  
69      /**
70       * Take a line from the log and attempt to strip it of metadata, such as
71       * severity, type and date.
72       * 
73       * @param line
74       *            Line from the log.
75       * @return The same line, stripped of metadata. Usually just the actual
76       *         information being logged.
77       */
78      String stripOfMetadata(final String line);
