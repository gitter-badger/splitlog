21   * This test has a weird name, due to the Abstract thing. But this is so that
22   * it's always run first in the alphabetical order, and therefore (when parallel
23   * surefire is enabled and more than 1 core is available) the run time of all
24   * tests will be minimized.

27  @RunWith(Parameterized.class)
28  public class MessageWritingTest extends DefaultFollowerBaseTest {
29  
30      private static final Logger LOGGER = LoggerFactory.getLogger(MessageWritingTest.class);
31      private static final int MESSAGES_TO_WRITE = 10;
32  
33      public MessageWritingTest(final LogWatchBuilder builder) {
34          super(builder);
35      }
36  
37      @Test
38      public void testWriteMessages() {
39          this.writeAndTest(false);
40      }
41  
42      @Test
43      public void testWriteMessagesAfterTerminated() {
44          this.writeAndTest(true);
45      }
46  
47      private void writeAndTest(final boolean closeBeforeWriting) {
48          final Follower follower = this.getLogWatch().startFollowing();
49          final List<String> messages = new LinkedList<String>();
50          for (int i = 0; i < MessageWritingTest.MESSAGES_TO_WRITE; i++) {
51              messages.add(this.getWriter().write(UUID.randomUUID().toString(), follower));
52          }
53          messages.remove(messages.size() - 1); // last message will not be
54          // written
55          if (closeBeforeWriting) {
56              this.getLogWatch().stopFollowing(follower);
57          }
58          try {
59              final File f = LogWriter.createTempFile();
60              MessageWritingTest.LOGGER.info("Will write into '{}'.", f);
61              follower.write(new FileOutputStream(f));
62              Assertions.assertThat(f).exists();
63              final List<String> lines = FileUtils.readLines(f, "UTF-8");
64              Assertions.assertThat(lines).isEqualTo(messages);
65          } catch (final Exception e) {
66              Assertions.fail("Couldn't write to file.");
67          } finally {
68              if (this.getLogWatch().isFollowedBy(follower)) {
69                  this.getLogWatch().stopFollowing(follower);
70              }
71          }
72      }
