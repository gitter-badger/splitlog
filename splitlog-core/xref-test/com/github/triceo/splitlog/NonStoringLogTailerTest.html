82          LogWatch watch = this.builder.build();
83          Assert.assertFalse("Log watch terminated immediately after starting.", watch.isTerminated());
84          LogTailer tailer1 = watch.startTailing();
85          Assert.assertFalse("Tailer terminated immediately after starting.", watch.isTerminated(tailer1));
86          LogTailer tailer2 = watch.startTailing();
87          Assert.assertTrue("Wrong termination result.", watch.terminateTailing(tailer1));
88          Assert.assertFalse("Wrong termination result.", watch.terminateTailing(tailer1));
89          Assert.assertFalse("Tailer terminated without termination.", watch.isTerminated(tailer2));
90          Assert.assertTrue("Tailer not terminated after termination.", watch.isTerminated(tailer1));
91          Assert.assertTrue("Wrong termination result.", watch.terminateTailing());
92          Assert.assertFalse("Wrong termination result.", watch.terminateTailing());
93          Assert.assertTrue("Tailer not terminated after termination.", watch.isTerminated(tailer2));
94          Assert.assertTrue("Log watch not terminated after termination.", watch.isTerminated());
