28          .isFalse();
29          final Follower follower1 = this.getLogWatch().startFollowing();
30          Assertions.assertThat(this.getLogWatch().isFollowedBy(follower1))
31          .as("Follower terminated immediately after starting.").isTrue();
32          final Follower follower2 = this.getLogWatch().startFollowing();
33          Assertions.assertThat(this.getLogWatch().stopFollowing(follower1)).as("Wrong termination result.").isTrue();
34          Assertions.assertThat(this.getLogWatch().stopFollowing(follower1)).as("Wrong termination result.").isFalse();
35          Assertions.assertThat(this.getLogWatch().isFollowedBy(follower2))
36          .as("Follower terminated without termination.").isTrue();
37          Assertions.assertThat(this.getLogWatch().isFollowedBy(follower1))
38          .as("Follower not terminated after termination.").isFalse();
39          Assertions.assertThat(this.getLogWatch().terminate()).as("Wrong termination result.").isTrue();
40          Assertions.assertThat(this.getLogWatch().terminate()).as("Wrong termination result.").isFalse();
41          Assertions.assertThat(this.getLogWatch().isFollowedBy(follower2))
42          .as("Follower not terminated after termination.").isFalse();
43          Assertions.assertThat(this.getLogWatch().isTerminated()).as("Log watch not terminated after termination.")
44          .isTrue();
45      }
