16  /**
17   * Has a sole responsibility of starting and stopping {@link Tailer} thread when
18   * told so by the {@link DefaultLogWatch}.
19   */
20  final class LogWatchTailingManager {
21  
22      private static final Logger LOGGER = LoggerFactory.getLogger(LogWatchTailingManager.class);
23  
24      private final int bufferSize;
25      private final long delayBetweenReads;
26      private final long delayedTailerStartInMilliseconds;
27      private final ScheduledExecutorService e = Executors.newScheduledThreadPool(1);
28      private final AtomicInteger numberOfTimesThatTailerWasStarted = new AtomicInteger(0);
29      private final boolean reopenBetweenReads, ignoreExistingContent;
30      private ScheduledFuture<?> tailer;
31      private final DefaultLogWatch watch;
32  
33      public LogWatchTailingManager(final DefaultLogWatch watch, final long delayBetweenReads,
34              final long delayForTailerStart, final boolean readFromEnd, final boolean reopenBetweenReads,
35              final int bufferSize) {
36          this.watch = watch;
37          this.delayBetweenReads = delayBetweenReads;
38          this.delayedTailerStartInMilliseconds = delayForTailerStart;
39          this.bufferSize = bufferSize;
40          this.reopenBetweenReads = reopenBetweenReads;
41          this.ignoreExistingContent = readFromEnd;
42      }
43  
44      public boolean isRunning() {
45          return this.tailer != null;
46      }
47  
48      /**
49       * Start the tailer on a separate thread. Only when a tailer is running can
50       * {@link Follower}s be notified of new {@link Message}s from the log.
51       * 
52       * @param needsToWait
53       *            Whether the start of the tailer needs to be delayed by
54       *            {@link #delayedTailerStartInMilliseconds} milliseconds, as
55       *            explained in
56       *            {@link LogWatchBuilder#getDelayBeforeTailingStarts()}.
57       * @return True if the start was scheduled, false if scheduled already.
58       */
59      public boolean start(final boolean needsToWait) {
60          if (this.isRunning()) {
61              LogWatchTailingManager.LOGGER.debug("Tailer already running, therefore not starting.");
62              return false;
63          }
64          final Tailer t = new Tailer(this.watch.getWatchedFile(), new LogWatchTailerListener(this.watch),
65                  this.delayBetweenReads, this.willReadFromEnd(), this.reopenBetweenReads, this.bufferSize);
66          final long delay = needsToWait ? this.delayedTailerStartInMilliseconds : 0;
67          this.tailer = this.e.schedule(t, delay, TimeUnit.MILLISECONDS);
68          if (this.numberOfTimesThatTailerWasStarted.getAndIncrement() == 0) {
69              LogWatchTailingManager.LOGGER.info("Scheduling log tailer for {} with delay of {} milliseconds.",
70                      this.watch, delay);
71          } else {
72              LogWatchTailingManager.LOGGER.info("Re-scheduling log tailer for {} with delay of {} milliseconds.",
73                      this.watch, delay);
74          }
75          return true;
76      }
77  
78      /**
79       * Stop the tailer thread, preventing any {@link Follower}s from receiving
80       * {@link Message}s.
81       * 
82       * @return True if stopped, false if never running.
83       */
84      public boolean stop() {
85          if (!this.isRunning()) {
86              LogWatchTailingManager.LOGGER.debug("Tailer not running, therefore not terminating.");
87              return false;
88          }
89          // forcibly terminate tailer
90          this.tailer.cancel(true);
91          this.tailer = null;
92          LogWatchTailingManager.LOGGER.info(
93                  "Terminated log tailer for {} as the last known Follower has just been terminated.", this.watch);
94          return true;
95      }
96  
97      /**
98       * Block the current thread until the tailer has finished starting. Delayed
99       * starts are a possibility with {@link #start(boolean)}.
100      */
101     public void waitUntilStarted() {
102         long remainingDelay = Long.MAX_VALUE;
103         // the tailer may have a delayed start; wait until it actually started
104         while (remainingDelay > 0) {
105             remainingDelay = this.tailer.getDelay(TimeUnit.MILLISECONDS) + 1;
106             if (remainingDelay < 0) {
107                 continue;
108             }
109             try {
110                 LogWatchTailingManager.LOGGER.debug("Will wait further {} milliseconds for tailer to be started.",
111                         remainingDelay);
112                 Thread.sleep(remainingDelay);
113             } catch (final InterruptedException e) {
114                 // do nothing
115             }
116         }
117     }
118 
119     private boolean willReadFromEnd() {
120         if (this.numberOfTimesThatTailerWasStarted.get() > 0) {
121             return true;
122         } else {
123             return this.ignoreExistingContent;
124         }
125     }
