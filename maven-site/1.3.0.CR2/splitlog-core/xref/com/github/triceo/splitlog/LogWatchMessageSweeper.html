3   import java.lang.ref.WeakReference;
4   import java.util.concurrent.ScheduledExecutorService;
5   
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;
8   
9   /**
10   * As followers are terminated via {@link #unfollow(Follower)} and then
11   * discarded by GC, some messages will become unreachable. (No followers point
12   * to them any longer.)
13   * 
14   * The point of this class is to review the situation and discard whatever
15   * messages are no longer reachable. Otherwise they'd be uselessly occupying
16   * memory.
17   * 
18   * This class is intended to be run periodically by
19   * {@link ScheduledExecutorService}.
20   */
21  final class LogWatchMessageSweeper implements Runnable {
22  
23      private static final Logger LOGGER = LoggerFactory.getLogger(LogWatchMessageSweeper.class);
24      /**
25       * The watchToSweep to log watch is weak, so that we prevent any possibility
26       * of watches from being reclaimed by the GC when they're no longer used by
27       * client code.
28       */
29      private final WeakReference<DefaultLogWatch> watchToSweep;
30  
31      public LogWatchMessageSweeper(final DefaultLogWatch watch) {
32          this.watchToSweep = new WeakReference<DefaultLogWatch>(watch);
33      }
34  
35      @Override
36      public void run() {
37          final DefaultLogWatch w = this.watchToSweep.get();
38          if (w == null) {
39              /*
40               * if the cleanup is still scheduled and the logwatch is already
41               * gone, terminate. the outside code should try to prevent this,
42               * though, by terminating the scheduler before disposing of the
43               * logwatch.
44               */
45              return;
46          }
47          final MessageStore messages = w.getMessageStore();
48          final int minId = w.getFirstReachableMessageId();
49          LogWatchMessageSweeper.LOGGER.debug(
50                  "Starting message sweep from log watch for file '{}'. First reachable message ID reportedly {}.",
51                  w.watchedFile, minId);
52          if (minId < 0) {
53              LogWatchMessageSweeper.LOGGER.info(
54                      "Sweeping all messages from log watch for file '{}' as none are reachable.", w.watchedFile);
55              messages.discardBefore(messages.getNextMessageId());
56              return;
57          } else if (messages.isEmpty()) {
58              LogWatchMessageSweeper.LOGGER.info("No messages in the log watch for file '{}'.", w.watchedFile);
59              return;
60          }
61          final int num = messages.discardBefore(minId);
62          LogWatchMessageSweeper.LOGGER.info("Swept {} messages from log watch for file '{}'.", num, w.watchedFile);
63      }
