3   import java.util.Collections;
4   import java.util.ConcurrentModificationException;
5   import java.util.LinkedList;
6   import java.util.List;
7   import java.util.concurrent.atomic.AtomicInteger;

9   /**
10   * Data storage for a particular {@link LogWatch}. Currently it is only a
11   * synchronization layer on top of a list, but in the future it may abstract
12   * away features like off-memory storage etc.
13   * 
14   * This class is thread-safe.
15   */
16  class MessageStore {
17  
18      private final List<Message> store = new LinkedList<Message>();
19      private final AtomicInteger lastMessageId = new AtomicInteger(-1);
20  
21      /**
22       * Add message to the storage.
23       * 
24       * @param msg
25       *            Message in question.
26       * @return ID of the message.
27       */
28      public synchronized int add(final Message msg) {
29          this.store.add(msg);
30          return this.lastMessageId.incrementAndGet();
31      }
32  
33      /**
34       * The latest ID that has been given out to a message.
35       * 
36       * @return -1 if no messages yet.
37       */
38      public int getLatestMessageId() {
39          return this.lastMessageId.get();
40      }
41  
42      /**
43       * The ID that will be given out to the message that goes through the very
44       * next {@link #add(Message)} call.
45       * 
46       * @return 0 if no messages yet.
47       */
48      public int getNextMessageId() {
49          return this.getLatestMessageId() + 1;
50      }
51  
52      /**
53       * This is synchronized in order to mutually exclude changing the store (by
54       * the also synchronized {@link #add(Message)} method) with getting the size
55       * of the store in this method. Otherwise
56       * {@link ConcurrentModificationException} may ensue.
57       * 
58       * @param startId
59       *            Least id, inclusive.
60       * @param endId
61       *            Greatest id, exclusive.
62       * @return A copy of the sub-list provided by the message store.
63       */
64      private synchronized List<Message> actuallyGetFromRange(final int startId, final int endId) {
65          return new LinkedList<Message>(this.store.subList(startId, endId));
66      }
67  
68      /**
69       * Return all messages with IDs in the given range.
70       * 
71       * @param startId
72       *            Least id, inclusive.
73       * @param endId
74       *            Greatest id, exclusive.
75       * @return Unmodifiable list containing those images.
76       */
77      public List<Message> getFromRange(final int startId, final int endId) {
78          if ((startId < 0) || (endId < 0)) {
79              throw new IllegalArgumentException("Message ID cannot be negative.");
80          } else if (endId <= startId) {
81              throw new IllegalArgumentException("Ending message ID must me larger than starting message ID.");
82          } else if (endId > this.getNextMessageId()) {
83              throw new IllegalArgumentException("Range end cannot be greater than the next message ID.");
84          }
85          return Collections.unmodifiableList(this.actuallyGetFromRange(startId, endId));
86      }
