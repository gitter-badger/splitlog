41      @Test(timeout = WaitingTest.TIMEOUT_MILLIS)
42      public void testConcurrentWaitingForMessages() throws InterruptedException, ExecutionException {
43          final HashMap<Future<Message>, String> tasks = new HashMap<Future<Message>, String>(WaitingTest.THREADS);
44          final Random random = new Random();
45          for (int i = 0; i < WaitingTest.THREADS; i++) {
46              // First thread always waits for the last message, the rest is
47              // random.
48              final String expectedValue = "<"
49                      + ((i == 0) ? String.valueOf(WaitingTest.TOTAL_MESSAGES - 1) : String.valueOf(random
50                              .nextInt(WaitingTest.TOTAL_MESSAGES))) + ">";
51              final Future<Message> task = this.es.submit(new Callable<Message>() {
52                  @Override
53                  public Message call() {
54                      final Follower follower = WaitingTest.this.getLogWatch().follow();
55                      return follower.waitFor(new MessageDeliveryCondition() {
56                          @Override
57                          public boolean accept(final Message evaluate, final MessageDeliveryStatus status) {
58                              return evaluate.getLines().get(0).endsWith(expectedValue);
59                          }
60                      }, WaitingTest.TIMEOUT_MILLIS / 2, TimeUnit.MILLISECONDS);
61                  }
62              });
63              tasks.put(task, expectedValue);
64          }
65          for (int i = 0; i < WaitingTest.TOTAL_MESSAGES; i++) {
66              this.getWriter().writeWithoutWaiting("<" + String.valueOf(i) + ">");
67          }
68          for (final Future<Message> task : tasks.keySet()) {
69              final Message accepted = task.get();
70              Assert.assertNotNull("Failed to accept message #" + tasks.get(task), accepted);
71          }
72          System.out.println("All messages collected.");
73      }
