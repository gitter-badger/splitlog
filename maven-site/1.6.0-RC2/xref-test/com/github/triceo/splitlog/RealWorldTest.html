31      /**
32       * Each ACCEPTED non-WARNING message will increase the count by 1
33       */
34      private static final MessageMeasure<Integer, LogWatch> ACCEPTED_ERROR = new MessageMeasure<Integer, LogWatch>() {
35  
36          @Override
37          public Integer initialValue() {
38              return 0;
39          }
40  
41          @Override
42          public Integer update(final MessageMetric<Integer, LogWatch> metric, final Message evaluate,
43              final MessageDeliveryStatus status, final LogWatch source) {
44              if (status != MessageDeliveryStatus.ACCEPTED) {
45                  return metric.getValue();
46              } else if (evaluate.getSeverity() != MessageSeverity.ERROR) {
47                  return metric.getValue();
48              } else {
49                  return metric.getValue() + 1;
50              }
51          }
52  
53      };

57       */

59  
60          @Override
61          public Integer initialValue() {
62              return 0;

64  
65          @Override
66          public Integer update(final MessageMetric<Integer, LogWatch> metric, final Message evaluate,

68              if (status != MessageDeliveryStatus.ACCEPTED) {
69                  return metric.getValue();

83      @Test
84      public void testRegular() {
85          // fill file with random garbage
86          final File tmp = LogWriter.createTempFile();
87          for (int i = 0; i < 1000; i++) {
88              try {
89                  FileUtils.write(tmp, UUID.randomUUID().toString(), "UTF-8", true);
90              } catch (final IOException e) {
91                  // ignore
92              }
93          }
94          // log watch will accept anything not DEBUG
95          final LogWatch watch = LogWatchBuilder.getDefault().watchedFile(tmp).ignoringPreexistingContent()
96                  .withStorageCondition(new SimpleMessageCondition() {
97  
98                      @Override
99                      public boolean accept(final Message evaluate) {
100                         return evaluate.getSeverity() != MessageSeverity.DEBUG;
101                     }
102 
103                 }).buildWith(new JBossServerLogTailSplitter());
104         // metric will only count everything not WARNING
105         final MessageMetric<Integer, LogWatch> nonWarning = watch.startMeasuring(RealWorldTest.ACCEPTED_NONWARNING,
106                 RealWorldTest.METRIC_ID);
107         final MessageMetric<Integer, LogWatch> errors = watch.startMeasuring(RealWorldTest.ACCEPTED_ERROR,
108                 RealWorldTest.METRIC_ID_2);
109         final Follower bothBatches = watch.startFollowing();
110         final int messagesInFirst = 66;
111         final Future<Message> expectation = nonWarning.expect(new MessageMetricCondition<Integer, LogWatch>() {
112 
113             @Override
114             public boolean accept(final MessageMetric<Integer, LogWatch> evaluate) {
115                 return evaluate.getValue() == (messagesInFirst - 1);
116             }
117 
118         });
119         try {
120             FileUtils.copyInputStreamToFile(RealWorldTest.class.getResourceAsStream("realworld.initial"),
121                     watch.getWatchedFile());
122         } catch (final IOException e) {
123             Assertions.fail("Failed copying file.", e);
124         }
125         // we expect 65 INFO messages...
126         final Message lastInfoInFirstBatch = DefaultFollowerBaseTest.wrapWaiting(expectation,
127                 RealWorldTest.BASE_TIMEOUT_SECONDS, TimeUnit.SECONDS);
128         Assertions.assertThat(lastInfoInFirstBatch).isNotNull();
129         // ... no ERRORs
130         Assertions.assertThat(errors.getValue()).isEqualTo(0);
131         // ... and 1 extra warning
132         Assertions.assertThat(bothBatches.getMessages()).hasSize(messagesInFirst);
133         /*
134          * last INFO is number 67, as even the 2 rejected messages get numbers;
135          * therefore 66 + 2 - 1, as we index messages from 0.
136          */
137         Assertions.assertThat(lastInfoInFirstBatch.getUniqueId()).isEqualTo(messagesInFirst + 1);
138         final Follower onlyLastBatch = watch.startFollowing();
139         // ##### now insert the second part #####
140         try {
141             final OutputStream tempFile = new FileOutputStream(watch.getWatchedFile(), true);
142             IOUtils.copy(RealWorldTest.class.getResourceAsStream("realworld.append"), tempFile);
143         } catch (final IOException ex) {
144             Assertions.fail("Couldn't append into the test file.", ex);
145         }
146         // expect 20 errors...
147         final Message lastError = DefaultFollowerBaseTest.wrapWaiting(
148                 errors.expect(new MessageMetricCondition<Integer, LogWatch>() {
149 
150                     @Override
151                     public boolean accept(final MessageMetric<Integer, LogWatch> evaluate) {
152                         return evaluate.getValue() == 20;
153                     }
154 
155                 }), RealWorldTest.BASE_TIMEOUT_SECONDS * 5, TimeUnit.SECONDS);
156         Assertions.assertThat(lastError).isNotNull();
157         Assertions.assertThat(lastError.getUniqueId()).isEqualTo(2599);
158         // wait for the last message
159         final int lastMessageId = 2672;
160         final Message lastMessageInSecondBatch = DefaultFollowerBaseTest.wrapWaiting(
161                 onlyLastBatch.expect(new MidDeliveryMessageCondition<LogWatch>() {
162 
163                     @Override
164                     public boolean accept(final Message evaluate, final MessageDeliveryStatus status, final LogWatch source) {
165                         return ((status == MessageDeliveryStatus.INCOMING) && (evaluate.getUniqueId() == lastMessageId));
166                     }
167 
168                 }));
169         Assertions.assertThat(lastMessageInSecondBatch).isNotNull();
170         Assertions.assertThat(lastMessageInSecondBatch.getUniqueId()).isEqualTo(lastMessageId);
171         Assertions.assertThat(bothBatches.getMessages()).hasSize(lastMessageId - 2);
172         /*
173          * the above difference of 2 are the 2 DEBUG messages not accepted at
174          * the beginning
175          */
176         Assertions.assertThat(onlyLastBatch.getMessages()).hasSize(bothBatches.getMessages().size() - messagesInFirst);
177         watch.terminate();
