44      @Test
45      public void testConcurrentExpectations() {
46          final HashMap<Future<Message>, String> tasks = new HashMap<Future<Message>, String>(ExpectationTest.THREADS);
47          final Random random = new Random();
48          for (int i = 0; i < ExpectationTest.THREADS; i++) {
49              // First thread always waits for the last message, the rest is
50              // random.
51              final String expectedValue = "<"
52                      + ((i == 0) ? String.valueOf(ExpectationTest.TOTAL_MESSAGES - 1) : String.valueOf(random
53                              .nextInt(ExpectationTest.TOTAL_MESSAGES))) + ">";
54              final Follower f = this.getLogWatch().startFollowing();
55              final Future<Message> task = f.expect(new MidDeliveryMessageCondition<LogWatch>() {
56  
57                  @Override
58                  public boolean
59                  accept(final Message evaluate, final MessageDeliveryStatus status, final LogWatch source) {
60                      return evaluate.getLines().get(0).endsWith(expectedValue);
61                  }
62  
63              });
64              tasks.put(task, expectedValue);
65          }
66          for (int i = 0; i < ExpectationTest.TOTAL_MESSAGES; i++) {
67              LogWriter.write(this.getLogWatch().getWatchedFile(), "<" + String.valueOf(i) + ">");
68          }
69          for (final Future<Message> task : tasks.keySet()) {
70              try {
71                  final Message accepted = task.get();
72                  Assertions.assertThat(accepted).as("Failed to accept message #" + tasks.get(task)).isNotNull();
73              } catch (final Exception e) {
74                  Assertions.fail("Failed to accept message #" + tasks.get(task), e);
75              }
76          }
77      }

79      @Test
80      public void testExpectationWithAction() {
81          final long timeout = 5000; // how long
82          final Follower f = this.getLogWatch().startFollowing();
83          final long startTime = System.nanoTime();
84          final Future<Message> future = f.expect(AllLogWatchMessagesAcceptingCondition.INSTANCE,
85                  new MessageAction<LogWatch>() {
86  
87              @Override
88              public void execute(final Message message, final LogWatch source) {
89                  try {
90                      // emulates some long-running action
91                      Thread.sleep(timeout);
92                  } catch (final InterruptedException e) {
93                      throw new IllegalStateException("Failed waiting.");
94                  }
95              }
96          });
97          LogWriter.write(f.getFollowed().getWatchedFile(), "test");
98          try {
99              future.get(); // should not return before the long-running
100             final long runTime = TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - startTime);
101             Assertions.assertThat(runTime).isGreaterThanOrEqualTo(timeout);
102         } catch (final Exception e) {
