10  /**
11   * Tailer's primary function is to allow users to work with their portion of the
12   * tailed log file. It provides means for a blocking wait for particular chunks,
13   * and can also send these chunks to output.
14   * 
15   */
16  public interface LogTailer {
17  
18      /**
19       * Retrieve messages that this tailer has been notified of, including tags
20       * 
21       * @return Messages we are aware of, in their original order.
22       */
23      List<Message> getMessages();
24  
25      /**
26       * Retrieve messages that this tailer has been notified of, if a certain
27       * condition holds true for them, including tags.
28       * 
29       * @param condition
30       *            The condition.
31       * @return Every message we are aware of, for which the condition holds
32       *         true.
33       */
34      List<Message> getMessages(final MessageCondition condition);
35  
36      boolean isTerminated();
37  
38      /**
39       * Mark the current location in the tail by a custom message. It is up to
40       * the implementors to decide whether or not a tag inserted at the same time
41       * twice is inserted twice, or the second tag overwrites the first.
42       * 
43       * @param tagLine
44       *            Text of the message.
45       * @return The tag message that was recorded.
46       */
47      Message tag(String tagLine);
48  
49      /**
50       * Will block until a message arrives, for which the condition is true.
51       * 
52       * @param condition
53       *            Condition that needs to be true for the method to unblock.
54       * @return Null if the method unblocked due to some other reason.
55       */
56      Message waitFor(MessageDeliveryCondition condition);
57  
58      /**
59       * Will block until a message arrives, for which the condition is true. If
60       * none arrives before the timeout, it unblocks anyway.
61       * 
62       * @param condition
63       *            Condition that needs to be true for the method to unblock.
64       * @return Null if the method unblocked due to some other reason.
65       */
66      Message waitFor(MessageDeliveryCondition condition, long timeout, TimeUnit unit);
67  
68      /**
69       * Will write to a stream the result of {@link #getMessages()}. Will close
70       * the stream.
71       * 
72       * @param stream
73       *            Target.
74       * @return True if written, false otherwise.
75       */
76      boolean write(final OutputStream stream);
77  
78      /**
79       * Will write to a stream the result of
80       * {@link #getMessages(MessageCondition)}. Will close the stream.
81       * 
82       * @param stream
83       *            Target.
84       * @param condition
85       *            The condition to pass to
86       *            {@link #getMessages(MessageCondition)}.
87       * @return True if written, false otherwise.
88       */
89      boolean write(final OutputStream stream, final MessageCondition condition);
