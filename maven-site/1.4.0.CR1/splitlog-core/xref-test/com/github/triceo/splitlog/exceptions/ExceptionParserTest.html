20      public static final Collection<String> parseIntoLines(final InputStream s) {
21          BufferedReader r = null;
22          try {
23              r = new BufferedReader(new InputStreamReader(s));
24              final Collection<String> lines = new ArrayList<String>();
25              String line = null;
26              while ((line = r.readLine()) != null) {
27                  lines.add(line);
28              }
29              return lines;
30          } catch (final Exception ex) {
31              return Collections.emptyList();
32          } finally {
33              IOUtils.closeQuietly(r);
34          }
35      }

37      @Test
38      public void testExceptionWithCauses() throws ExceptionParseException {
39          final List<ExceptionLine> lines = new ArrayList<ExceptionLine>(
40                  ExceptionParser.INSTANCE.parse(ExceptionParserTest.parseIntoLines(this.getClass().getResourceAsStream(
41                          "exception-with-causes.txt"))));
42          Assertions.assertThat(lines.size()).isEqualTo(48); // one item per line
43                                                             // in the file
44          // verify the initial cause
45          final CauseLine firstLine = (CauseLine) lines.get(0);
46          Assertions.assertThat(firstLine.getClassName()).isEqualTo("org.jboss.qa.bpms.rest.wb.RequestFailureException");
47          Assertions.assertThat(firstLine.getMessage()).isEqualTo("java.net.ConnectException: Connection refused");
48          // verify one native method
49          final StackTraceLine stackTraceLine = (StackTraceLine) lines.get(7);
50          Assertions.assertThat(stackTraceLine.getSource()).isEqualTo(Source.NATIVE);
51          Assertions.assertThat(stackTraceLine.getMethodName()).isEqualTo("sun.reflect.NativeMethodAccessorImpl.invoke0");
52          Assertions.assertThat(stackTraceLine.getClassSource()).isEqualTo("na");
53          Assertions.assertThat(stackTraceLine.getClassSourceVersion()).isEqualTo("1.7.0_51");
54          // verify one causedBy
55          final CauseLine rootCause = (CauseLine) lines.get(23);
56          Assertions.assertThat(rootCause.getClassName()).isEqualTo("java.net.ConnectException");
57          Assertions.assertThat(rootCause.getMessage()).isEqualTo("Connection refused");
58          // verify last bit
59          final StackTraceEndLine endLine = (StackTraceEndLine) lines.get(47);
60          Assertions.assertThat(endLine.getHowManyOmmitted()).isEqualTo(27);
