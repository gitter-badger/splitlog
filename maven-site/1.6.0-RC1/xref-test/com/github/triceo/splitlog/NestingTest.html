15  /**
16   * This test has a weird name, due to the Abstract thing. But this is so that
17   * it's always run first in the alphabetical order, and therefore (when parallel
18   * surefire is enabled and more than 1 core is available) the run time of all
19   * tests will be minimized.
20   *
21   */
22  @RunWith(Parameterized.class)
23  public class NestingTest extends DefaultFollowerBaseTest {
24  
25      public NestingTest(final LogWatchBuilder builder) {
26          super(builder);
27      }
28  
29      @Test
30      public void testNesting() {
31          final String message1 = "test1";
32          final String message2 = "test2";
33          final String message3 = "test3";
34          final String message4 = "test4";
35          final String message5 = "test5";
36          final Follower follower = this.getLogWatch().startFollowing();
37          // make sure the messages are received by the first follower
38          this.getWriter().write(message1, follower);
39          String result = this.getWriter().write(message2, follower);
40          Assertions.assertThat(result).isEqualTo(message2);
41          Assertions.assertThat(follower.getMessages().size()).isEqualTo(1);
42          // start a second follower, send some messages
43          final Follower nestedFollower = this.getLogWatch().startFollowing();
44          result = this.getWriter().write(message3, follower);
45          result = this.getWriter().write(message4, follower);
46          Assertions.assertThat(result).isEqualTo(message4);
47          this.getLogWatch().stopFollowing(nestedFollower);
48          // send another message, so the original follower has something extra
49          Assertions.assertThat(nestedFollower.isStopped()).isTrue();
50          result = this.getWriter().write(message5, follower);
51          // and make sure that the original follower has all messages
52          final List<Message> messages = new LinkedList<Message>(follower.getMessages());
53          Assertions.assertThat(messages.size()).isEqualTo(4);
54          Assertions.assertThat(messages.get(0).getLines().get(0)).isEqualTo(message1);
55          Assertions.assertThat(messages.get(1).getLines().get(0)).isEqualTo(message2);
56          Assertions.assertThat(messages.get(2).getLines().get(0)).isEqualTo(message3);
57          Assertions.assertThat(messages.get(3).getLines().get(0)).isEqualTo(message4);
58          // and the nested follower has only the two while it was running
59          final List<Message> nestedMessages = new LinkedList<Message>(nestedFollower.getMessages());
60          Assertions.assertThat(nestedFollower.getMessages().size()).isEqualTo(2);
61          Assertions.assertThat(nestedMessages.get(0).getLines().get(0)).isEqualTo(message2);
62          Assertions.assertThat(nestedMessages.get(1).getLines().get(0)).isEqualTo(message3);
63      }
