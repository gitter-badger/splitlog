23  @RunWith(Parameterized.class)
24  public class TaggingTest extends DefaultFollowerBaseTest {

30              return false;

33      }

35      public TaggingTest(final LogWatchBuilder builder) {
36          super(builder);
37      }
38  
39      @Test
40      public void testTag() {
41          final String message1 = "test";
42          final String message2 = "test7";
43          final String message3 = "test11";
44          final String tag0 = "tag1";
45          final String tag1 = "tag1";
46          final String tag2 = "tag2";
47          final Follower follower = this.getLogWatch().startFollowing();
48          follower.tag(tag0);
49          String result = this.getWriter().write(message1, follower);
50          Assertions.assertThat(result).isEqualTo(message1);
51          /*
52           * when this tag is being created, the first message is already
53           * instantiated. therefore, it will come after this message.
54           */
55          follower.tag(tag1);
56          result = this.getWriter().write(message2, follower);
57          Assertions.assertThat(result).isEqualTo(message2);
58          // ditto
59          follower.tag(tag2);
60          result = this.getWriter().write(message3, follower);
61          Assertions.assertThat(result).isEqualTo(message3);
62          // make sure all messages are present with the default condition
63          List<Message> messages = new LinkedList<Message>(follower.getMessages());
64          Assertions.assertThat(messages.size()).isEqualTo(5);
65          Assertions.assertThat(messages.get(0).getLines().get(0)).isEqualTo(tag0);
66          Assertions.assertThat(messages.get(1).getLines().get(0)).isEqualTo(message1);
67          Assertions.assertThat(messages.get(2).getLines().get(0)).isEqualTo(tag1);
68          Assertions.assertThat(messages.get(3).getLines().get(0)).isEqualTo(message2);
69          Assertions.assertThat(messages.get(4).getLines().get(0)).isEqualTo(tag2);
70          // make sure just the tags are present witn nothing-accepting condition
71          messages = new LinkedList<Message>(follower.getMessages(new NothingAcceptingMessageCondition()));
72          Assertions.assertThat(messages.size()).isEqualTo(3);
73          Assertions.assertThat(messages.get(0).getLines().get(0)).isEqualTo(tag0);
74          Assertions.assertThat(messages.get(1).getLines().get(0)).isEqualTo(tag1);
75          Assertions.assertThat(messages.get(2).getLines().get(0)).isEqualTo(tag2);
76      }
