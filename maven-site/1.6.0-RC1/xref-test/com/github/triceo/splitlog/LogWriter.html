25  /**
26   * Write a message to a given log file.
27   *
28   * This class is designed in such a way that it would be literally impossible
29   * for two threads to attempt to write the same file at the same time.
30   *
31   */
32  public class LogWriter {
33  
34      /**
35       * The longer this delay, the less likely that a message will be missed by
36       * the subsequent waitFor() calls.
37       */
38      private static final long DELAY_BEFORE_WRITE_MILLIS = 100;
39      private static final Logger LOGGER = LoggerFactory.getLogger(LogWriter.class);
40      /**
41       * This provides exclusive access to each file through
42       * {@link #forFile(File)}.
43       */
44      private static final Map<File, LogWriter> WRITERS = new HashMap<File, LogWriter>();
45      public static synchronized LogWriter forFile(final File f) {
46          if (!LogWriter.WRITERS.containsKey(f)) {
47              LogWriter.WRITERS.put(f, new LogWriter(f));
48          }
49          return LogWriter.WRITERS.get(f);
50      }
51      private final ScheduledExecutorService e = Executors.newScheduledThreadPool(1);
52      private boolean isDisposed = false;
53  
54      private final File target;
55  
56      private LogWriter(final File target) {
57          this.target = target;
58      }
59  
60      public synchronized void dispose() {
61          LogWriter.WRITERS.remove(this.target);
62          this.e.shutdownNow();
63          this.isDisposed = true;
64      }
65  
66      /**
67       * Writes a line to the log and waits until the follower receives it.
68       *
69       * @param line
70       *            Message to write.
71       * @param follower
72       *            Follower to wait for the message.
73       * @return The line that was written, or null otherwise.
74       */
75      public String write(final String line, final Follower follower) {
76          this.writeDelayed(line);
77          // wait until the last part of the string is finally present
78          Message result;
79          try {
80              result = follower.expect(new MidDeliveryMessageCondition<LogWatch>() {
81  
82                  @Override
83                  public boolean
84                      accept(final Message evaluate, final MessageDeliveryStatus status, final LogWatch source) {
85                      final String lastLine = evaluate.getLines().get(evaluate.getLines().size() - 1);
86                      final String textStr[] = line.split("\\r?\\n");
87                      return (textStr[textStr.length - 1].trim().equals(lastLine.trim()));
88                  }
89  
90              }).get();
91              return result.getLines().get(result.getLines().size() - 1);
92          } catch (final Exception e) {
93              throw new IllegalStateException("No message received in time.", e);
94          }
95      }
96  
97      /**
98       * Write message to a file on background.
99       *
100      * This method will schedule the write operation, but it will not actually
101      * be ececuted until {@link #DELAY_BEFORE_WRITE_MILLIS} milliseconds later.
102      * This is so that the subsequent
103      * {@link CommonFollower#waitFor(MidDeliveryMessageCondition)}s have a chance
104      * to be registered.
105      *
106      * @param line
107      *            Message to write.
108      */
109     private synchronized void writeDelayed(final String line) {
110         LogWriter.LOGGER.info("Delayed write of '{}' to {} scheduled.", line, LogWriter.this.target);
111         this.e.schedule(new Callable<Boolean>() {
112 
113             @Override
114             public Boolean call() throws Exception {
115                 LogWriter.LOGGER.info("Delayed write of '{}' to {} executing now.", line, LogWriter.this.target);
116                 return LogWriter.this.writeNow(line);
117             }
118 
119         }, LogWriter.DELAY_BEFORE_WRITE_MILLIS, TimeUnit.MILLISECONDS);
120     }
121 
122     public synchronized boolean writeNow(final String line) {
123         if (this.isDisposed) {
124             LogWriter.LOGGER.info("Not writing '{}' into {} as the writer is already disposed.", line, this.target);
125             return false;
126         }
127         BufferedWriter w = null;
128         try {
129             w = new BufferedWriter(new FileWriter(this.target, true));
130             w.write(line);
131             w.newLine();
132             w.flush();
133             LogWriter.LOGGER.info("Written '{}' into {}.", line, this.target);
134         } catch (final IOException ex) {
135             LogWriter.LOGGER.warn("Failed writing '{}' into {}.", line, this.target, ex);
136             return false;
137         } finally {
138             IOUtils.closeQuietly(w);
139         }
140         return true;
141     }
