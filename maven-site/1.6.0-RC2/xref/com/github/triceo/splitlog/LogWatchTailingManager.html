18  /**
19   * Has a sole responsibility of starting and stopping {@link Tailer} thread when
20   * told so by the {@link DefaultLogWatch}.
21   */
22  final class LogWatchTailingManager {
23  
24      private static final Executor EXECUTOR = Executors.newCachedThreadPool(new SplitlogThreadFactory("tails"));
25  
26      private static final Logger LOGGER = SplitlogLoggerFactory.getLogger(LogWatchTailingManager.class);
27      private final int bufferSize;
28      private final long delayBetweenReads;
29      private final AtomicLong numberOfTimesThatTailerWasStarted = new AtomicLong(0);
30      private final boolean reopenBetweenReads, ignoreExistingContent;
31      private SplitlogTailer tailer;
32      private final DefaultLogWatch watch;
33  
34      public LogWatchTailingManager(final DefaultLogWatch watch, final LogWatchBuilder builder) {
35          this.watch = watch;
36          this.delayBetweenReads = builder.getDelayBetweenReads();
37          this.bufferSize = builder.getReadingBufferSize();
38          this.reopenBetweenReads = builder.isClosingBetweenReads();
39          this.ignoreExistingContent = !builder.isReadingFromBeginning();
40      }
41  
42      public synchronized boolean isRunning() {
43          return (this.tailer != null);
44      }
45  
46      /**
47       * Start the tailer on a separate thread. Only when a tailer is running can
48       * {@link Follower}s be notified of new {@link Message}s from the log.
49       *
50       * @return True if the start was scheduled, false if scheduled already.
51       */
52      public synchronized boolean start() {
53          if (this.isRunning()) {
54              return false;
55          }
56          this.tailer = new SplitlogTailer(this.watch.getWatchedFile(), new LogWatchTailerListener(this.watch),
57                  this.delayBetweenReads, this.willReadFromEnd(), this.reopenBetweenReads, this.bufferSize);
58          LogWatchTailingManager.EXECUTOR.execute(this.tailer);
59          final long start = System.nanoTime();
60          this.tailer.waitUntilStarted();
61          final long duration = System.nanoTime() - start;
62          LogWatchTailingManager.LOGGER.debug("It took {} ms for the tailing to actually start.",
63                  TimeUnit.NANOSECONDS.toMillis(duration));
64          final long iterationNum = this.numberOfTimesThatTailerWasStarted.incrementAndGet();
65          LogWatchTailingManager.LOGGER.info("Tailing #{} started for {}.", iterationNum, this.watch);
66          return true;
67      }
68  
69      /**
70       * Stop the tailer thread, preventing any {@link Follower}s from receiving
71       * {@link Message}s.
72       *
73       * @return True if stopped, false if never running.
74       */
75      public synchronized boolean stop() {
76          if (!this.isRunning()) {
77              LogWatchTailingManager.LOGGER.debug("Tailer not running, therefore not terminating.");
78              return false;
79          }
80          // forcibly terminate tailer
81          this.tailer.stop();
82          this.tailer = null;
83          LogWatchTailingManager.LOGGER.info("Terminated tailing #{} for {}.",
84                  this.numberOfTimesThatTailerWasStarted.get(), this.watch);
85          return true;
86      }
87  
88      private synchronized boolean willReadFromEnd() {
89          if (this.numberOfTimesThatTailerWasStarted.get() > 0) {
90              return true;
91          } else {
92              return this.ignoreExistingContent;
93          }
94      }
