49      private final CountDownLatch latch = new CountDownLatch(1);

56      protected AbstractExpectation(final AbstractExpectationManager<S, C> manager, final C condition,
57          final MessageAction<S> action) {
58          if (manager == null) {
59              throw new IllegalArgumentException("Must provide manager.");
60          } else if (condition == null) {
61              throw new IllegalArgumentException("Must provide condition.");
62          }
63          this.action = action;
64          this.manager = manager;
65          this.blockingCondition = condition;
66      }

68      /**
69       * Will start the wait for the condition to become true.
70       */
71      @Override
72      public Message call() {
73          if (this.latch.getCount() == 0) {
74              throw new IllegalStateException("Expectation already triggered.");
75          } else if (this.stash != null) {
76              /*
77               * in case the condition has been triggered before this has been
78               * called, we retrieve the message from the stash.
79               */
80              final Message msg = this.stash;
81              this.stash = null;
82              return msg;
83          }
84          try {
85              AbstractExpectation.LOGGER.info("Thread blocked waiting for message to pass condition {}.",
86                      this.getBlockingCondition());
87              this.latch.await();
88              this.manager.unsetExpectation(this); // don't notify again
89              AbstractExpectation.LOGGER.info("Thread unblocked.");
90              if (this.actionFuture != null) {
91                  AbstractExpectation.LOGGER.info("Executing user action.");
92                  try {
93                      // wait for the future action to finish
94                      this.actionFuture.get();
95                  } catch (final Exception e) {
96                      AbstractExpectation.LOGGER.warn("User action ended abruptly.", e);
97                  }
98              }
99              return this.stash;
100         } catch (final InterruptedException e) {
101             return null;
102         } finally {
103             this.manager.unsetExpectation(this); // in case await() throws
104             AbstractExpectation.LOGGER.info("Expectation processing finished.");
105         }
106     }
107 
108     protected C getBlockingCondition() {
109         return this.blockingCondition;
110     }

114      *

123     protected abstract boolean isAccepted(final Message msg, final MessageDeliveryStatus status, final S source);

125     @Override
126     public synchronized void messageReceived(final Message msg, final MessageDeliveryStatus status, final S source) {
127         AbstractExpectation.LOGGER.info("Notified of message '{}' in state {} from {}.", msg, status, source);
128         // check if the user code accepts the message
129         if (!this.isAccepted(msg, status, source)) {
130             return;
131         }
132         AbstractExpectation.LOGGER.debug("Condition passed by message '{}' in state {} from {}.", msg, status, source);
133         if (this.stash == null) {
134             /*
135              * if call() hasn't been called yet, and the condition has already
136              * triggered, stash the message for retrieval when call() is
137              * actually called.
138              */
139             AbstractExpectation.LOGGER.debug("Stashing the message.");
140             this.stash = msg;
141             this.latch.countDown();
142             if (this.action == null) {
143                 return;
144             }
145             try {
146                 this.actionFuture = AbstractExpectation.EXECUTOR.submit(new Callable<Void>() {
147 
148                     @Override
149                     public Void call() throws Exception {
150                         AbstractExpectation.this.action.execute(msg, source);
151                         return null;
152                     }
153 
154                 });
155             } catch (final Throwable t) {
156                 AbstractExpectation.LOGGER.info("Caught Throwable while executing user action {}.", this.action, t);
157             }
158         } else {
159             AbstractExpectation.LOGGER.debug("Message already stashed.");
160         }
161     }
