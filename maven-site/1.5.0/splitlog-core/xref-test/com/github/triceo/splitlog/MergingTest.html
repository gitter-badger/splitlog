28          this.getWriter().write("test3", f3);
29          final MergingFollower mf2 = mf.mergeWith(f3);
30          Assertions.assertThat(mf2).isNotEqualTo(mf);
31          Assertions.assertThat(mf.getMessages().size()).isEqualTo(2);
32          Assertions.assertThat(mf2.getMessages().size()).isEqualTo(2);
33          // remove both followers from first merge, verify results
34          mf.separate(f);
35          Assertions.assertThat(mf.getMessages().size()).isEqualTo(2);
36          Assertions.assertThat(mf.isStopped()).isFalse();
37          Assertions.assertThat(this.getLogWatch().stopFollowing(f2)).isTrue();
38          Assertions.assertThat(mf.isStopped()).isTrue(); // no followers are
39                                                             // following
40          Assertions.assertThat(mf.getMessages().size()).isEqualTo(2);
41          mf.separate(f2);

43                                                             // following
44          Assertions.assertThat(mf.getMessages().size()).isEqualTo(0);
45          // none of these changes should have affected the second merge
