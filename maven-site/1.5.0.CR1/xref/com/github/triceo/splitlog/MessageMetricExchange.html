16  final class MessageMetricExchange<T extends Number> implements MessageListener<MessageSource> {
17  
18      private static final Logger LOGGER = LoggerFactory.getLogger(MessageMetricExchange.class);
19  
20      private MessageMetricCondition<T> messageBlockingCondition = null;
21      private final Exchanger<Message> messageExchanger = new Exchanger<Message>();
22      private final MessageMetric<T> metric;
23  
24      public MessageMetricExchange(final MessageMetric<T> metric) {
25          this.metric = metric;
26      }
27  
28      @Override
29      public void messageReceived(final Message msg, final MessageDeliveryStatus status, final MessageSource source) {
30          MessageMetricExchange.LOGGER.info("Notified of message '{}' in state {} from {}.", msg, status, source);
31          if (this.messageBlockingCondition == null) {
32              MessageMetricExchange.LOGGER.debug("Not blocked.");
33              // this does nothing with the message
34              return;
35          }
36          // check if the user code accepts the message
37          if (!this.messageBlockingCondition.accept(this.metric)) {
38              return;
39          }
40          MessageMetricExchange.LOGGER
41                  .debug("Condition passed by message '{}' in state {} from {}.", msg, status, source);
42          this.messageBlockingCondition = null;
43          try {
44              this.messageExchanger.exchange(msg);
45          } catch (final InterruptedException e) {
46              MessageMetricExchange.LOGGER.warn("Failed to notify Metric of message '{}' in state {} from {}.", msg,
47                      status, source, e);
48          }
49      }
50  
51      /*
52       * The method is synchronized, therefore no thread will be able to overwrite
53       * an already set condition. That condition will later be unset by the
54       * notify*() method calls from the tailing thread.
55       */
56      public synchronized Message waitForMessage(final MessageMetricCondition<T> condition, final long timeout,
57          final TimeUnit unit) {
58          this.messageBlockingCondition = condition;
59          try {
60              MessageMetricExchange.LOGGER.info("Thread blocked waiting for message to pass condition {}.", condition);
61              if (timeout < 0) {
62                  return this.messageExchanger.exchange(null);
63              } else {
64                  return this.messageExchanger.exchange(null, timeout, unit);
65              }
66          } catch (final TimeoutException e) {
67              return null;
68          } catch (final InterruptedException e) {
69              return null;
70          } finally { // just in case
71              MessageMetricExchange.LOGGER.info("Thread unblocked.");
72              this.messageBlockingCondition = null;
73          }
74      }
