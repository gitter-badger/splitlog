280     /**
281      * Specify the delay between attempts to read the file.
282      * 
283      * @param length
284      *            Length of time.
285      * @param unit
286      *            Unit of that length.
287      * @return This.
288      */
289     public LogWatchBuilder withDelayBetweenReads(final int length, final TimeUnit unit) {
290         if (length < 1) {
291             throw new IllegalArgumentException("The length of time between reads must be at least 1.");
292         }
293         switch (unit) {
294             case NANOSECONDS:
295                 if (length < (1000 * 1000)) {
296                     throw new IllegalArgumentException("The length of time between reads must amount to at least 1 ms.");
297                 }
298                 break;
299             case MICROSECONDS:
300                 if (length < 1000) {
301                     throw new IllegalArgumentException("The length of time between reads must amount to at least 1 ms.");
302                 }
303                 break;
304             default:
305                 // every other unit is more than 1 ms by default
306         }
307         this.delayBetweenReads = unit.toMillis(length);
308         return this;
309     }
