9        * Purpose of this test is to make sure that, if the property is not set,
10       * the logging is disabled by default.
11       */
12      @Test
13      public void testDefaultValue() {
14          // the default will be set by surefire to "on"; reset that
15          final String defaultValue = System.getProperty(SplitlogLoggerFactory.LOGGING_PROPERTY_NAME,
16                  SplitlogLoggerFactory.OFF_STATE);
17          Assertions.assertThat(defaultValue).isEqualTo(SplitlogLoggerFactory.ON_STATE);
18          System.getProperties().remove(SplitlogLoggerFactory.LOGGING_PROPERTY_NAME);
19          // and now check the default default :-)
20          final boolean isEnabled = SplitlogLoggerFactory.isLoggingEnabled();
21          /*
22           * but before that, restore the original; if the assertion at the end
23           * fails, we want the original state for the rest of the tests
24           */
25          System.setProperty(SplitlogLoggerFactory.LOGGING_PROPERTY_NAME, defaultValue);
26          Assertions.assertThat(isEnabled).isFalse();
27      }

29      @Test
30      public void testLogging() {
31          final boolean defaultValue = SplitlogLoggerFactory.isLoggingEnabled();
32          SplitlogLoggerFactory.enableLogging();
33          Assertions.assertThat(SplitlogLoggerFactory.isLoggingEnabled()).isTrue();
34          Assertions.assertThat(SplitlogLoggerFactory.getMessagesSinceLastStateChange()).isEqualTo(0);
35          SplitlogLoggerFactory.getLogger(InternalLoggingTest.class).error("This will be logged.");
36          final long logged = SplitlogLoggerFactory.getMessagesSinceLastStateChange();
37          SplitlogLoggerFactory.resetLoggingToDefaultState();
38          Assertions.assertThat(SplitlogLoggerFactory.isLoggingEnabled()).isEqualTo(defaultValue);
39          Assertions.assertThat(logged).isEqualTo(1);
40      }
