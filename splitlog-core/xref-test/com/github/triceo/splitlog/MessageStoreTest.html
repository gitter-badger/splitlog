100         final int MESSAGE_ID_2 = MESSAGE_ID_1 + 1;

102         // single-message capacity for easy testing

105         final Message msg1 = new Message("test");
106         store.add(msg1);
107         Assert.assertEquals(MESSAGE_ID_1, store.getFirstMessageId());
108         // discarding the first message
109         final Message msg2 = new Message("test2");
110         store.add(msg2);
111         Assert.assertEquals(MESSAGE_ID_2, store.getFirstMessageId());
112         List<Message> postDiscard = store.getFrom(MESSAGE_ID_2);
113         Assert.assertEquals(1, postDiscard.size());
114         Assert.assertSame(msg2, postDiscard.get(0));
115         // discarding the second message
116         final Message msg3 = new Message("test3");
117         store.add(msg3);
