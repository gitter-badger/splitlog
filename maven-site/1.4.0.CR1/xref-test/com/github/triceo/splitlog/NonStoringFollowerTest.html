30      private void writeAndTest(final boolean closeBeforeWriting) {
31          final Follower follower = this.getLogWatch().follow();
32          final List<String> messages = new LinkedList<String>();
33          for (int i = 0; i < NonStoringFollowerTest.MESSAGES_TO_WRITE; i++) {
34              messages.add(this.getWriter().write(UUID.randomUUID().toString(), follower));
35          }
36          messages.remove(messages.size() - 1); // last message will not be
37                                                // written
38          if (closeBeforeWriting) {
39              this.getLogWatch().unfollow(follower);
40          }
41          try {
42              final File f = File.createTempFile("splitlog-", ".log");
43              NonStoringFollowerTest.LOGGER.info("Will write into '{}'.", f);
44              follower.write(new FileOutputStream(f));
45              Assertions.assertThat(f).exists();
46              final List<String> lines = FileUtils.readLines(f, "UTF-8");
47              Assertions.assertThat(lines).isEqualTo(messages);
48          } catch (final Exception e) {
49              Assertions.fail("Couldn't write to file.");
50          } finally {
51              if (this.getLogWatch().isFollowedBy(follower)) {
52                  this.getLogWatch().unfollow(follower);
53              }
54          }
55      }

67      @Test
68      public void testGetMessages() {
69          final String message1 = "test";
70          final String message2part1 = "test1";
71          final String message2part2 = "test2";
72          final String message3part1 = "test3";
73          final String message3part2 = "test4";
74          final Follower follower = this.getLogWatch().follow();
75          // test simple messages
76          String result = this.getWriter().write(message1, follower);
77          Assertions.assertThat(result).isEqualTo(message1);
78          result = this.getWriter().write(message1, follower);
79          Assertions.assertThat(result).isEqualTo(message1);
80          result = this.getWriter().write(message2part1 + "\n" + message2part2, follower);
81          Assertions.assertThat(result).isEqualTo(message2part2);
82          result = this.getWriter().write(message3part1 + "\r\n" + message3part2, follower);
83          Assertions.assertThat(result).isEqualTo(message3part2);
84          // now validate the results
85          final List<Message> messages = new LinkedList<Message>(follower.getMessages());
86          Assertions.assertThat(messages.size()).isEqualTo(5);
87          Assertions.assertThat(messages.get(1).getLines().get(0)).isEqualTo(message1);
88          Assertions.assertThat(messages.get(2).getLines().get(0)).isEqualTo(message2part1);
89          Assertions.assertThat(messages.get(3).getLines().get(0)).isEqualTo(message2part2);
90          Assertions.assertThat(messages.get(4).getLines().get(0)).isEqualTo(message3part1);
91          // final part of the message, message3part2, will remain unflushed
92          this.getLogWatch().unfollow(follower);
93          Assertions.assertThat(follower.isFollowing()).isFalse();
94      }

128     @Test
129     public void testNesting() {
130         final String message1 = "test1";
131         final String message2 = "test2";
132         final String message3 = "test3";
133         final String message4 = "test4";
134         final String message5 = "test5";
135         final Follower follower = this.getLogWatch().follow();
136         // make sure the messages are received by the first follower
137         this.getWriter().write(message1, follower);
138         String result = this.getWriter().write(message2, follower);
139         Assertions.assertThat(result).isEqualTo(message2);
140         Assertions.assertThat(follower.getMessages().size()).isEqualTo(1);
141         // start a second follower, send some messages
142         final Follower nestedFollower = this.getLogWatch().follow();
143         result = this.getWriter().write(message3, follower);
144         result = this.getWriter().write(message4, follower);
145         Assertions.assertThat(result).isEqualTo(message4);
146         this.getLogWatch().unfollow(nestedFollower);
147         // send another message, so the original follower has something extra
148         Assertions.assertThat(nestedFollower.isFollowing()).isFalse();
149         result = this.getWriter().write(message5, follower);
150         // and make sure that the original follower has all messages
151         final List<Message> messages = new LinkedList<Message>(follower.getMessages());
152         Assertions.assertThat(messages.size()).isEqualTo(4);
153         Assertions.assertThat(messages.get(0).getLines().get(0)).isEqualTo(message1);
154         Assertions.assertThat(messages.get(1).getLines().get(0)).isEqualTo(message2);
155         Assertions.assertThat(messages.get(2).getLines().get(0)).isEqualTo(message3);
156         Assertions.assertThat(messages.get(3).getLines().get(0)).isEqualTo(message4);
157         // and the nested follower has only the two while it was running
158         final List<Message> nestedMessages = new LinkedList<Message>(nestedFollower.getMessages());
159         Assertions.assertThat(nestedFollower.getMessages().size()).isEqualTo(2);
160         Assertions.assertThat(nestedMessages.get(0).getLines().get(0)).isEqualTo(message2);
161         Assertions.assertThat(nestedMessages.get(1).getLines().get(0)).isEqualTo(message3);
162     }

178         Assertions.assertThat(this.getLogWatch().terminate()).as("Wrong termination result.").isTrue();
179         Assertions.assertThat(this.getLogWatch().terminate()).as("Wrong termination result.").isFalse();
180         Assertions.assertThat(this.getLogWatch().isFollowedBy(follower2))
181                 .as("Follower not terminated after termination.").isFalse();
182         Assertions.assertThat(this.getLogWatch().isTerminated()).as("Log watch not terminated after termination.")
183                 .isTrue();
