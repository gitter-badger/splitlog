14  /**
15   * Prepares an instance of {@link LogWatch}. Unless overriden by the user, the
16   * instance will have the following properties:
17   *
18   * <dl>
19   * <dt>Default gating condition</dt>
20   * <dd>{@link SplitlogMessagesRejectingCondition}.</dd>
21   * <dt>Default storage condition</dt>
22   * <dd>{@link AllFollowerMessagesAcceptingCondition}.</dd>
23   * </dl>
24   *
25   */
26  final public class DefaultLogWatchBuilder extends LogWatchBuilder {
27  
28      private static final Logger LOGGER = SplitlogLoggerFactory.getLogger(DefaultLogWatchBuilder.class);
29  
30      public DefaultLogWatchBuilder() {
31          super();
32          this.withGateCondition(SplitlogMessagesRejectingCondition.INSTANCE);
33          this.withStorageCondition(AllFollowerMessagesAcceptingCondition.INSTANCE);
34      }
35  
36      @Override
37      public LogWatch build() {
38          return this.buildWith(new SimpleTailSplitter());
39      }
40  
41      @Override
42      public Follower buildFollowing() {
43          return this.buildFollowingWith(new SimpleTailSplitter());
44      }
45  
46      @Override
47      public Follower buildFollowingWith(final TailSplitter splitter) {
48          if (splitter == null) {
49              throw new IllegalArgumentException("A splitter must be provided.");
50          }
51          return this.buildWith(splitter).startFollowing();
52      }
53  
54      @Override
55      public LogWatch buildWith(final TailSplitter splitter) {
56          if (splitter == null) {
57              throw new IllegalArgumentException("A splitter must be provided.");
58          } else if ((splitter instanceof SimpleTailSplitter)
59                  && (this.getGateCondition() instanceof SplitlogMessagesRejectingCondition)) {
60              DefaultLogWatchBuilder.LOGGER
61              .warn("Using default TailSplitter with default gating condition. All messages will pass through gate, as the TailSplitter will not provide all the necessary information.");
62          }
63          return new DefaultLogWatch(this.getFileToWatch(), splitter, this.getCapacityLimit(), this.getGateCondition(),
64                  this.getStorageCondition(), this.getDelayBetweenReads(), this.getDelayBetweenSweeps(),
65                  !this.isReadingFromBeginning(), this.isClosingBetweenReads(), this.getReadingBufferSize());
66      }
